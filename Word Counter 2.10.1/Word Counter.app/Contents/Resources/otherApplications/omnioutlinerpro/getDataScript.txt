tell application "OmniOutliner Professional"	set thePath to POSIX path of (path to desktop) as string	set fileName to ".hiddenWordCounterFile.rtf"	set posixPathString to (thePath & fileName)	set posixPath to POSIX path of (posixPathString)	export front document to posixPath as "OORTFPboardType"	set textToReturn to do shell script "textutil -stdout -convert txt " & quoted form of posixPath	do shell script "rm " & quoted form of posixPath	return textToReturnend tell
